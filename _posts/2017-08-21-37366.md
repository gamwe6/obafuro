df = df.sort_values(by=u'日付').reset_index(drop=True)

for short_day, long_day in types:

    # 移動平均

    dead   = (over_s_l != over_s_l.shift(1)) &amp; (over_s_l == False)

    # 列 追加

    #  -1：デッドクロス

    df[col_name] = [x+y*-1 for x,y in zip(golden, dead)]
