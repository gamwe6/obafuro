<pre><code>True &amp;&amp; _ = _
False &amp;&amp; _ = False
</code></pre>

<pre><code>(bb) :: Bool -&gt; a -&gt; b

</code></pre>

<pre><code>enzan.hs:1:1: error:

1 | (bb) :: Bool -&gt; a -&gt; b || Bool
  | ^^^^
</code></pre>
