    // ファイルを読み出し
    public String readFile(String file) {
        String text = null;

        try {
            FileInputStream in = openFileInput(file);
            BufferedReader reader = new BufferedReader(new InputStreamReader(in, "UTF-8"));
            String str = "";
            String tmp;
            while ((tmp = reader.readLine()) != null) {
                str = str + tmp + "\n";
            }
            text = str;
            reader.close();
        } catch (IOException e) {
            e.printStackTrace();
        }
        return text;
    }
}
