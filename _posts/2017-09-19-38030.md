<pre><code>text = open("test.txt",encoding="utf-8")
t = text.read()
sentence = t.split("\n")

for tt in sentence:
    tt = tt.replace(",","") #　,を削除
    tt = tt.replace(".","") #　.を削除
    tt = tt.replace(";","") #　;を削除
    tt = tt.replace("?","") #  ?を削除
    tt = tt.replace("!","") #  !を削除
    words = tt.split() 
   #小文字でもOKにしたい
    swords = [] 
    for s in words:             
        ss = s.lower()    
        swords.append(ss)


k = input("検索ワードを入力")
if words.count(k) &gt;= 1:
    print("合計で",words.count(k),"回出てきます。") 
    while words.count(k) &gt;= 1: 
        print(words.index(k),"番目") 
　      words.remove(k)

elif swords.count(k) &gt;= 1:
    print("合計で",swords.count(k),"回出てきます") 
    while swords.count(k) &gt;= 1:
        print(swords.index(k),"番目") 
        swords.remove(k)

else:
    print("お探しの単語は見つかりませんでした。")
text.close()
</code></pre>
