    // 元のデータ長＋カンマの数で領域を確保
    std::string str = data;

    // カンマを挿入する
    typedef std::vector&lt;int&gt;::iterator VecIt;
    VecIt comma_index = seperater.begin();
    const VecIt end = seperater.end();
    std::string::iterator itr = str.begin();

    while (comma_index != end)
    {
        itr += *comma_index;
        // はみ出たらその場で終了する。

        itr = str.insert(itr, ',');
        ++comma_index;
        ++itr;
    }

    // 最後のカンマを消す
    if (*(itr - 1) == ',')
    {
        str.replace(itr - 1, itr, "");
    }
    std::cout &lt;&lt; str;
    return str;

}
