### hiddenパラメータに格納できるようにエンコード
sub unpack_param {
my ($str) = @_;

$str =~ s/([^\w.\/\-@ ])/'%'.unpack('H2',$1)/eg;

return $str;
}

### エンコードしたhiddenパラメータをデコード
sub pack_param {
my ($str) = @_;

$str =~ s/%([A-Fa-f0-9]{2})/pack('H2', $1)/eg;

return $str;
}
