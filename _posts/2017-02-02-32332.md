<pre><code>SELECT 　　 id
           ,kaisu
           ,henkou
           ,furigana 
           ,sex
           ,nengo
           ,birthday
           ,idno
  FROM tableA AS A,
  (
  SELECT id as Bid
      ,MAX(kaisu) as Bkaisu
  FROM tableA as B
  GROUP BY Bid
  ) as B
  where A.id = B.Bid and A.kaisu = B.Bkaisu
order by A.id
</code></pre>
