<pre><code>class Hoge
  attr_accessor :foo

  def initialize(value)
    @foo = Foo.new(value)
  end
end

class Foo
  attr_accessor :value

  def initialize(value)
    @value = value
  end
end

values = [
  Hoge.new(1),
  Hoge.new(2)
]


p values.map{|hoge| hoge.foo.value} #=&gt; 問題なし
p values.map(&amp;:foo.value) #=&gt; エラー
</code></pre>

<blockquote>
  <p>undefined method `value' for :foo:Symbol (NoMethodError)</p>
