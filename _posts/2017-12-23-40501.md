<p>【Phone.javaの内容】</p>

<pre><code>@Documented
@Constraint(validatedBy=PhoneValidator.class)
@Target({ElementType.METHOD,ElementType.FIELD})
@Retention(RetentionPolicy.RUNTIME)
@ReportAsSingleViolation
public @interface Phone {
    String message() default "電話番号の入力が不正です。";
    Class&lt;?&gt;[] groups() default {};
    Class&lt;? extends Payload&gt;[] payload() default{};
    boolean onlyNumber() default false;
}
</code></pre>

<p>【PhoneValidator .javaの内容】</p>

<pre><code>public class PhoneValidator implements ConstraintValidator&lt;Phone, String&gt; {
    private boolean onlyNumber = true;

    @Override
    public void initialize(Phone phone){
        this.onlyNumber=phone.onlyNumber();
    }

    @Override
    public boolean isValid(String input,ConstraintValidatorContext cxt){
        if(input == null){
            return false;
        }
        if(this.onlyNumber){
            return input.matches("[0-9]*");
        }
        else{
            return input.matches("[0-9()-]*");
        }
    }
}
</code></pre>

<pre><code>public class MyData {

    @Column(length=200,nullable=true)

    @Column(nullable=true)
