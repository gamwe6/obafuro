<p>スライドと拡大させたいJavaです。</p>

<pre><code>kekka=0;
$(function(){
  $("#pics ul li a").mouseup(function(e){
      no=$("#pics ul li a").index(this);        
      $("#d_none ul li a").eq(no).css("display","block").addClass("big");
  });

  $("#slider").slider({
    min:0,
    max:130*12-500+20,      
    change:function(e,v){
        //alert(v.value);
        var kekka=v.value*-1;
        $("#pics ul").css("left",kekka);
    }
 });

});
</code></pre>
