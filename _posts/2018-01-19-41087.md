<pre><code># サイトの関数をmarume.ceilから、int_ceilに変更。
def int_ceil(src,range):
    return ((int)(src / range) + 1) * range

def main():
    week = int(input())
    debt = 100000
    risi = int_ceil(int(debt*0.05) * week,10000)
    ans = debt + risi
    print(ans)

if __name__ == "__main__":
    main()
