<p>/* basetypes.c */    </p>

<pre><code>#include &lt;stdlib.h&gt;
#include &lt;stdio.h&gt;
#include "basetypes.h"

PCB *pcb_new (int start, int required)
{
  PCB *pcb = (PCB *)malloc(sizeof(PCB));
  pcb-&gt;next = NULL;
  pcb-&gt;pid = 0;            
  pcb-&gt;state = READY;
  pcb-&gt;waiting_time = 0;
  pcb-&gt;executed_time = 0;
  pcb-&gt;start_time = start;     
  pcb-&gt;required_time = required; 
  return pcb;
} 

void pcb_enqueue(PCB *pcb, PCB **q)
{
  if (*q == pcb) {
    printf("Failed!!!\n");
    exit(0);
  }
  pcb-&gt;next = NULL;
  if (! *q)
    *q = pcb;
  else {
    PCB *p = *q;
    while (p-&gt;next != NULL) {
      p = p-&gt;next;
    }
    if (p-&gt;next == pcb) {
      printf("Failed!!!\n");
      exit(0);
    }
    p-&gt;next = pcb;
  }
}

PCB *pcb_dequeue(PCB **q)
{
  PCB *pcb;
  if (! *q)
    return NULL;
  PCB *p = (*q);
  pcb = (*q)-&gt;next;
  (*q) = pcb;
  return p;
}

void pcb_print_list(PCB *top)
{
  if (top == NULL) {
    printf("[]");
  }
  else {
    while(top != NULL){
      printf("[pid%d,%d,%d]", top-&gt;pid, top-&gt;start_time, top-&gt;required_time);
      top=top-&gt;next;
      if (top)
    printf(", ");
    }
  }
  printf("\n");
}

int main (void)
{
  /* 変数の宣言 */
  PCB *p1 = pcb_new(0, 1);
  PCB *p2 = pcb_new(1, 2);
  PCB *p3 = pcb_new(2, 3);
  PCB *p4 = pcb_new(3, 4);
  PCB *p5 = pcb_new(4, 5);
  PCB *x1;
  /* pcb_enqueue */
  pcb_enqueue(p2, &amp;p1);
  pcb_enqueue(p3, &amp;p1);
  pcb_enqueue(p4, &amp;p1);
  pcb_enqueue(p5, &amp;p1);
  pcb_print_list(p1);
  /* pcb_dequeue */
  x1 = pcb_dequeue(&amp;p1);
  pcb_print_list(x1);
  pcb_print_list(p1);

  return 0;
}
