<p>・node.jsのhttpサーバ</p>

<pre><code> var http = require('http');
    var server = http.createServer(function (req, res) {
    res.writeHead(
        200,

            'Content-Type':'text/html'

    );// END writeHead
    res.end('&lt;h1&gt;Hello Node&lt;/h1&gt;');
});//END createServer
server.listen(8080);
console.log('server started on xxxx');
</code></pre>
