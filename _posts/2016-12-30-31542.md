<pre><code>*,
*:after,
*:before{
    -webkit-box-sizing: border-box;
       -moz-box-sizing: border-box;
            box-sizing: border-box;
    margin: 0;
    padding: 0;
    -webkit-overflow-scrolling: touch;
}
html,
body{

    overflow-x: hidden;

    height: 100%;
