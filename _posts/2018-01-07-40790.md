<p>Aizu Onlineジャッジの<a href="http://judge.u-aizu.ac.jp/onlinejudge/description.jsp?id=0005&amp;lang=jp" rel="nofollow noreferrer">最大公約数と最小公倍数の問題</a>をPython3で解いていますが、「Runtime Error」（はやく言えばスクリプト言語単位でのエラーですね）と出てしまいます。<br>
どこに「Runtime Error」の原因があるのか教えてくれないでしょうか。</p>

<pre><code>class GCD():
    def __init__(self,a,b):
        self.a = a
        self.b = b

    def swap(self):
        temp = self.a
        self.a = self.b
        self.b = temp

    def gcd(self):
        if self.b &lt; self.a:
            self.swap()

        temp = self.b % self.a

        if temp == 0:
            ans = self.a
            return ans

        ans = self.b % temp

        return ans


    def lcm(self):
        gcd = self.gcd()
        temp = self.a * self.b
        ans = temp // gcd
        return gcd,ans

    def print(self):
        num = self.lcm()
        print("{0} {1}".format(num[0],num[1]))

def main():
    data = []

    while 1:
        try:
            n = input().split()

            a = int(n[0])
            b = int(n[1])

            data.append(GCD(a, b))

        except EOFError:
            break

    for array in data:
        array.print()

if __name__ == "__main__":
    main()
