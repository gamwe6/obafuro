            rs.close();
            stmt.close();
            conn.close();
            while (rs.next()) {
                int prefCd = rs.getInt("pref_cd");

                pref.setAreaCd(areaCd);
                pref.setContCd(contCd);
                pref.setHelzCd(helzCd);

            System.out.println("例外発生:" + e);
        }
        return pref;

    }
