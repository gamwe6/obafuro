コードは</p>

<pre><code>class Circle{
    double x;
    double y;
    double radius;

        this (x, 0, 1);
    }
    Circle(double x, double y){
        this(x, y, 1);
    }
    Circle(double x , double y, double radius){
        this.x      = x;

}
class CircleThis{
