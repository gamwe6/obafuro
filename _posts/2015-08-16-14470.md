<pre><code>#include &lt;unistd.h&gt;
#include &lt;fcntl.h&gt;
int
main(void)
{
    int fdi, fdo;
    char b[1];
    fdi = open("in", O_RDONLY);
    fdo = open("out", O_WRONLY);
    while (0 &lt; read(fdi, b, 1)) {
        write(fdo, b, 1);
    }
    close(fdi);
    close(fdo);
}
