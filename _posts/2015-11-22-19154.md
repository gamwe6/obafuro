<p>gulpとjade, compassを使ってフロントのビルドをしています。</p>

<pre><code>── jade
│   │   ├── _partial
│   │   │   ├── _banners.jade
│   │   │   ├── _breadcrumb.jade
│   │   │   ├── _fixed_header.jade
│   │   │   ├── _footer.jade
│   │   │   ├── _head.jade
│   │   │   ├── _header.jade
│   │   │   ├── _lists.jade
│   │   │   ├── _maps.jade
│   │   │   ├── _recommend.jade
│   │   │   ├── _second_banner.jade
│   │   │   ├── _sidebar.jade
│   │   │   ├── _top_banner.jade
│   │   │   ├── _top_search.jade
│   │   │   └── _tweets.jade
│   │   └── index.jade
</code></pre>

<p>という構成で<code>index.html</code>のみ出力をしたいのですが、<code>_partial</code>フォルダも一緒に出力されてしまいます。</p>

<p>以下gulpfile.jsになります。</p>

<pre><code>'use strict';
var gulp = require('gulp');
var compass = require('gulp-compass');
var jade = require('gulp-jade');
var plumber = require('gulp-plumber');

gulp.task('jade', function() {
    gulp.src('./app/jade/**/*.jade')
    .pipe(plumber())
    .pipe(jade({
        pretty: true
    }))
    .pipe(gulp.dest('./dist/html'));
});

gulp.task('compass', function() {
    gulp.src('compass', function() {
        gulp.src('.app/sass/**/*.scss')
        .pipe(plumber())
        .pipe(compass({
            config_file: 'config.rb',
            comments: false,
            css: './dist/css/',
            sass: './app/sass/'
        }));
    });
});

gulp.task('watch', ['jade','compass'], function() {
    gulp.watch('./app/jade/**/*.jade', ['jade']);
    gulp.watch('./app/sass/**/*.scss', ['compass']);
});

gulp.task('default', ['jade', 'compass']);
</code></pre>
