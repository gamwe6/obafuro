<p>いつもお世話になっております。<br>
SKSceneの背景を表示したいのですが、何故か？真っ黒のままになってしまいます。<br>
翔泳社から出ている「SpriteKitではじめる2Dゲームプログラミング」と同じコードなのですが、<br>
どこがいけなんでしょうか？<br>
お解りになる方、教えていただけますか？</p>

<p>下記がコードです。</p>

<p>GameScene.swift</p>

<pre><code>import SpriteKit

class GameScene: SKScene {

    override func didMoveToView(view: SKView) {
        let backGround = SKSpriteNode(imageNamed:"haikei.png")
        backGround.position = CGPoint(x: self.size.width * 0.5, y: self.size.height * 0.5)
        backGround.size = self.size
        self.addChild(backGround)
    }
}
</code></pre>

<p>GameViewController.swift</p>

<pre><code>import UIKit import SpriteKit

class GameViewController: UIViewController {

    override func viewDidLoad() {
        super.viewDidLoad()

        let scene = GameScene()
        let view = self.view as! SKView
        view.showsFPS = true
        view.showsNodeCount = true
        scene.size = view.frame.size

        view.presentScene(scene)
    }
}
</code></pre>
