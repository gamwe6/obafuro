<pre><code>#!/bin/bash
ipaddr="10.0.1.1"
count=10

echo "Ping to: $ipaddr"
echo "Ping count: $count"
echo "Ping average[ms]:"

#Pingコマンドを実行し、結果をファイルに出力
ping -c $count $ipaddr &gt; ping.txt

#time=??ms部分をsedで取り出し（|¥で改行）
sed -n "s/^.*time=¥(.*¥) ms/¥1/p" ping.txt |¥
#行数(NR)で割って、平均を求める
awk '{sum+=$1} END{print sum/NR}' &gt; ping2.txt
</code></pre>
