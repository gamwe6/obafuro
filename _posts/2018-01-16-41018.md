<pre><code>cnt = 0
while True:
    for tweet in data:

        print(tweet["created_at"])#ツイート日時

        maxid = int(tweet["id"]) - 1

    if len(data) == 0:

    url = "https://api.twitter.com/1.1/search/tweets.json?count=100&amp;lang=ja&amp;q=" + word + "&amp;max_id=" + str(maxid)
