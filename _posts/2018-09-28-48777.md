<pre><code>#include &lt;stdio.h&gt;
int main()
{
    int end = 0;
    char d[5];
    int digits = 0;
    d[digits++] = digits;
    printf("digits=%d\n",digits);
    printf("d[0]=%d\n",d[0]);

    do{
        printf("if enter 1, end.");
        scanf("%d",&amp;end);
    }while (end = 0);

    return 0;
}
</code></pre>

<p>---実行結果---<br>
digits=1<br>
d[0]=1<br>
if enter 1, end.</p>

<p>---僕の理想の実行結果---<br>
digits=1<br>
d[0]=0　←ここが0<br>
if enter 1, end.</p>
