<pre><code>&lt;h2&gt;switch文2&lt;/h2&gt;&lt;br&gt;

    String cha= "あ";String flash = " ";

    switch (cha){

        flash = "日本語";
        break;

        flash = "英語";

        flash = "error";
        break;
