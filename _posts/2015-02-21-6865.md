def replace_uri(s)
    str = s.dup
    uri_reg = URI.regexp(%w[http https])
    str.gsub!(uri_reg) {"#{$&amp;}"}
    "#{$&amp;}"
end

NOGIZAKA_BLOG = "http://blog.nogizaka46.com"

def expand_url(url)
  uri = url.kind_of?(URI) ? url : URI.parse(url)

    r['Location'] || uri.to_s
