<pre><code>from sympy import *
var('s,t')
myLT3=integrate(exp(-s*t), (t, 0, 1))
print(myLT3)
u = symbols('u', positive=True)
print(myLT3.subs(s, u).subs(u, s))
#Piecewise((1, Eq(s, 0)), (1/s - exp(-s)/s, True))
#1/s - exp(-s)/s
</code></pre>
