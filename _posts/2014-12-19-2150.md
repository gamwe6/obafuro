import cv2

cap = cv2.VideoCapture(0)

while(cap.isOpend()):
    ret,frame = cap.read()
    if ret == True:

        out.write(frame)

        cv2.imshow('frame',frame)
        if cv2.waitKey(1) &amp; 0xFF == ord("q"):
