var pObj = new Promise(function (resolve, reject) {
  reject('テスト');
})

</code></pre>

  <p>Promise {[[PromiseStatus]]: "rejected", [[PromiseValue]]: "テスト"}<br>

<p><strong>疑問＆質問</strong><br>
