int KeyBord_Get(int keycode)
{

    char tmpkey[256];
    GetHitKeyStateAll(tmpkey);
    for (int i = 0; i &lt; 256; i++)
    {
        if (tmpkey[i] != 0)
        {
            key[i]++;
        }
        else
        {
            key[i] = 0;
        }

    }

    return key[keycode];
}
