};

<p>A.cpp</p>

<pre><code>#include "A.h"

    return ptr;
}

void A::init() {
    _b = B::create();
}
</code></pre>

<p>B.h</p>

<pre><code>#include &lt;memory&gt;

<p>B.cpp</p>

<pre><code>#include "B.h"

std::unique_ptr&lt;B&gt; B::create() {
        std::unique_ptr&lt;B&gt; ptr(new B);
        return ptr;
}
</code></pre>

<p>main.cpp</p>

<pre><code>#include "A.h"

int main(void) {
    auto a = A::create();
    return 0;
}
</code></pre>
