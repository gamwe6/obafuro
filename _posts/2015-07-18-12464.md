<p>Socket.ioで双方向通信のWebアプリを開発中です。<br>
現在、サーバ側からのemitはできるのですが、クライアントからのemitが送信できない状態です。</p>

<p>server.js</p>

<pre><code>var fs = require('fs'),
    socketio = require('socket.io'),
    path = require('path');

var server = require('http').createServer();
server.on('request', doRequest);
server.listen(1234);

var io = socketio.listen(server);
io.sockets.on('connection', function(socket) {

    socket.emit('connect');

    socket.on('msg', function(data) {
        console.log(data);
    });
});

function doRequest(req, res) {
    switch(path.extname(req.url)) {
    case '':
        fs.readFile('./index.html', 'UTF-8',
            function (err, data) {
                res.writeHead(200, {'Content-Type': 'text/html'});
                res.write(data);
                res.end();
            }
        );
        break;
    case '.css':
        fs.readFile('.'+req.url, 'UTF-8',
            function (err, data) {
                res.writeHead(200, {'Content-Type': 'text/css'});
                res.write(data);
                res.end();
            }
        );
        break;
    case '.js':
        fs.readFile('.'+req.url, 'UTF-8',
            function (err, data) {
                res.writeHead(200, {'Content-Type': 'application/javascript'});
                res.write(data);
                res.end();
            }
        );
        break;
    }
}
</code></pre>

<p>index.html</p>

<pre><code>&lt;!DOCTYPE html&gt;
&lt;html&gt;
&lt;head&gt;
    &lt;meta http-equiv="content-type" content="text/html"&gt;
    &lt;meta charset="UTF-8"&gt;

    &lt;link rel="stylesheet" type="text/css" href="style.css"&gt;

    &lt;script&gt;
    var socket = io.connect('http://localhost:1234');

    socket.on('connect', function() {
        socket.emit('msg', 'connect!');

    function send() {
        var msg = $('#msg').val();
        $('#msg').val('');

    &lt;form&gt;
        &lt;textarea id="msg" rows="4" cols="50"&gt;&lt;/textarea&gt;

    &lt;/form&gt;
    &lt;div id="container"&gt;
    &lt;/div&gt;
&lt;/body&gt;
&lt;/html&gt;
</code></pre>

<p>index.html読み込み時、サーバ側からのemitによってクライアント側の以下のイベントが発火します。</p>

<pre><code>socket.on('connect', function() {
    socket.emit('msg', 'connect!');
    console.log('connect!');
});
</code></pre>

<p>Chromeのコンソールには「connect!」と表示されますが、同関数内のsocket.emitは実行されず、サーバ側でlistenしている</p>

<pre><code>socket.on('msg', function(data) {
    console.log(data);
});
</code></pre>

<p>こちらのイベントが発火しません。</p>

<p>また、textareaに文字を入力して送信ボタンを押しても、同様にsend()関数のsocket.emit部分のみ実行できていないです。</p>
