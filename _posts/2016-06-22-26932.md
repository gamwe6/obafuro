&lt;uses-permission android:name="android.permission.GET_TASKS" /&gt;

&lt;uses-permission

&lt;uses-permission

&lt;uses-permission
    android:name="android.permission.READ_EXTERNAL_STORAGE"
    android:maxSdkVersion="18" /&gt;

&lt;application...
</code></pre>
