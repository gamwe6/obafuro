void insert(int x)
{
    if(size==sA-1)
    {
        int* oldpA = pA;
        sA = sA*2;
        if((pA=(int*)malloc(sizeof(int)*sA))==0) exit(1);
        memcpy(pA, oldpA, sizeof(int)*(size+1));
        free(oldpA);
    }
    size++;
    pA[size]=x;

}
