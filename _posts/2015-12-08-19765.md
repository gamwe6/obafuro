<p>しかし、これはでスタックしてしまいます。<br>
このようなプログラムをスタックせずに実行するにはどうすればいいのでしょうか。</p>

<p>プログラム</p>

<pre><code>package main

import "fmt"
import "runtime"

func main() {
    runtime.GOMAXPROCS(runtime.NumCPU())
    msgCh := make(chan string)
    count := 0.0
    for i := 0; i &lt; 255; i++ {
        go first(msgCh)
    }

L:
    for {
        select {
        case msg := &lt;-msgCh:
            count++
            fmt.Printf("%f:%s\n", count, msg)
            if count == 4294967296 {
                break L
            }
        default:
        }
    }

}

func first(msgCh chan string) {
    for i := 0; i &lt; 255; i++ {
        go second(msgCh)
    }
}

func second(msgCh chan string) {
    for i := 0; i &lt; 255; i++ {
        go third(msgCh)
    }
}

func third(msgCh chan string) {
    for i := 0; i &lt; 255; i++ {
        msgCh &lt;- "Hello"
    }
}
</code></pre>
