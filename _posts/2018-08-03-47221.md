def sort(N):

    flag = ~(N &amp; (N - 1))
    if flag != -1:
        return None

    result = np.zeros(N, dtype=np.int64)

    result[0] = 0
    result[1] = N / 2

    result[N / 2] = 1
    result[N / 2 + 1] = N / 2 + 1

    T = n  = N / 2
    oldN = n
    oldCat = 0
    cat = 1

    count = 1
    judge = 0
    next = 2

    linspace = np.array([i for i in range(N)])

    while(n != 2):

        n /= 2
        cat *= 2

        judge += cat

        normal = 2 ** (count + 1))
        for i in range(cat / 2):

            j = 1 + i * 2
            index = j * N / normal

            result[index] = next
            result[index + 1] = next + T

            result[index + T] = next + 1
            result[index + T + 1] = next + T + 1

            next += 2

        count += 1
        oldN = n 

    return result
