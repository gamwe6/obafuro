$(function() {
$.getJSON("maps.json", function(markers){

  var title = new Array();
  var lat = new Array();
  var lng = new Array();
  var text = new Array();
  for (var i = 0; i &lt; markers.length; i++) {
    title[i] = markers[i][0];
    lat[i] = markers[i][1];
    lng[i] = markers[i][2];
    text[i] = markers[i][3];
 }

 });
