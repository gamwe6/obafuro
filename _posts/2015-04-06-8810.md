// Loadingイメージ表示関数
function dispLoading(msg) {

    // 画面表示メッセージ
    var dispMsg = "";

    // 引数が空の場合は画像のみ
    if (msg != "") {

        dispMsg = "&lt;div class='loadingMsg'&gt;" + msg + "&lt;/div&gt;";
    }

    // ローディング画像が表示されていない場合のみ表示
    if ($("#loading").size() == 0) {

        $("body").append("&lt;div id='loading'&gt;" + dispMsg + "&lt;/div&gt;");
    }
}

// Loadingイメージ削除関数
function removeLoading() {

    // Loadingイメージを消す
    $("#loading").remove();
}
