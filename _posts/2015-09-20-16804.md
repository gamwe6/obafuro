# タグの属性にアクセス

from xml.dom import minidom

# sample.xmlファイルを読み込む
xdoc = minidom.parse("sample.xml")

# recipe タグの0番目の要素を取得
recipe_element = xdoc.getElementsByTagName("recipe")[0]

# name 属性を取得して表示
print(recipe_element.getAttribute("name"))

print("の父親は")

# name 属性を変更
recipe_element.setAttribute("name", "悟飯")

# 再度 name 属性を表示
print(recipe_element.getAttribute("name"))
