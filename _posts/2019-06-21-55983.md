<p>実行プログラム</p>

<pre><code>import heapq
import math
a = [6, 3, 2, 4, 5, 7, 1, 8]
heapq.heapify(a)

num = int(math.log2(len(a)))

depth_list = []
for i in range(num+1):
    depth_list.append(i)

print('depth_list')
print(depth_list)


depth = int(math.log2(len(a)))
element= [2**x for x in range(depth)] + [len(a) - (2**depth - 1)]
print('element')
print(element)

total = 0
for j in range (len(depth_list)):
    total += depth[j]*element[j]
    # 0*1 + 1*2 + 2*4 + 3*1

print('tolal')
print(total)
</code></pre>
